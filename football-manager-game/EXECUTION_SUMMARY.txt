🚀 USAGE EXAMPLE

// Initialize Core (loads all dependencies)
FMG.Core.initialize();

// Convert legacy state to Core
const adapter = FMG.Core.adapter;
let gameState = adapter.toCore();

// Advance week
const result = FMG.Core.engine.advanceWeek(gameState, {
  weekSeed: 12345
});

// Sync back to legacy
adapter.syncFromCore(result.gameState);

// Save to storage
const repo = new FMG.Core.Repository.GameStateRepository();
await repo.save("save1", result.gameState);
